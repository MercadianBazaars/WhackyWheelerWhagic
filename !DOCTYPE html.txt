    <meta charset="UTF-8">

    <title>Whacky Wheeler Whagic</title>
    <style>
        body { font-family: Arial, sans-serif; text-align: center; padding: 20px; }
        #cardImage { width: 250px; height: auto; margin: 20px 0; }
        #guessInput { padding: 10px; margin: 10px; }
        #feedback { font-weight: bold; margin-top: 10px; }
    </style>
</head>
<body>
    <h1>Whacky Wheeler Whagic</h1>
    <p>Guess the Magic: The Gathering card!</p>
    <img id="cardImage" src="" alt="Magic Card" />
    <br>
    <input type="text" id="guessInput" placeholder="Enter card name...">
    <button onclick="checkGuess()">Submit Guess</button>
    <p id="feedback"></p>
    
    <script>
        let cardData = null;

        function fetchRandomCard() {
            fetch("https://api.scryfall.com/cards/random?q=set:lea OR set:leb OR set:2ed OR set:3ed OR set:4ed OR set:5ed OR set:6ed OR set:7ed OR set:8ed OR set:9ed OR set:10e OR set:m10 OR set:m11 OR set:m12 OR set:m13 OR set:m14 OR set:m15 OR set:ori OR set:dmu")
            .then(response => response.json())
            .then(data => {
                cardData = data;
                document.getElementById("cardImage").src = data.image_uris ? data.image_uris.normal : "https://via.placeholder.com/250";
            })
            .catch(error => console.error("Error fetching card:", error));
        }

        function checkGuess() {
            const userGuess = document.getElementById("guessInput").value.trim().toLowerCase();
            if (!cardData) {
                document.getElementById("feedback").textContent = "Error: No card data loaded!";
                return;
            }

            if (userGuess === cardData.name.toLowerCase()) {
                document.getElementById("feedback").textContent = "✔ Correct!";
                setTimeout(fetchRandomCard, 2000); // Load a new card after 2 seconds
            } else {
                document.getElementById("feedback").textContent = "✘ Incorrect, try again!";
            }
        }

        // Fetch the first random card when the page loads
        fetchRandomCard();
    </script>
</body>
</html>
